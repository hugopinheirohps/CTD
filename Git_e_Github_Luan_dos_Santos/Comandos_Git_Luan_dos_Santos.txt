git init // cria um repositório

git config --global user.name nomeUsuario // agrega seu nome ao repositório

git config --global user.email nomeUsuario@email.com // agrega seu email ao repositório

git remote add origin https://github.com/DH/RepoRemoto // aponta os arquivos para um repositório remoto

git clone https:// Copia os arquivos remotos para a máquina

git add // adiciona todos os arquivos

git status // mostra o estado dos arquivos

git push origin main // envia as mudanças para o repositório remoto

git branch “nomeDaBranch” // Cria uma nova branch (cópia dos arquivos)

git checkout “nomeDaBranch” // Muda para a nova branch

git push origin nomeDaBranch // Envia a nova branch para o GitHub

git merge nomeDaBranch // mescla a branch com a master (precisa trocar para a master antes)

git branch -d nomeDaBranch // Exclui a branch localmente

git push origin  - -delete nomeDaBranch
